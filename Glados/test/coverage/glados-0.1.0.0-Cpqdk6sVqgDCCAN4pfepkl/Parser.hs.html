    8 module Parser (
    9     parseChar,
   10     parseAnyChar,
   11     parseOr,
   12     parseAnd,
   13     parseSome,
   14     parseMany,
   15     parseUInt,
   16     parseInt,
   17     parsePair,
   18     parseList, Parser(..)) where
   19 
   20 import Data.Char ()
   21 
   22 newtype Parser a = Parser { runParser :: String -> Maybe (a, String) }
   23 
   24 parseChar :: Char -> Parser Char
   25 parseChar a = Parser $ \input ->
   26   case input of
   27     (b:bx) | a == b -> Just (a, bx)
   28     (_:_) -> Nothing
   29     [] -> Nothing
   30 
   31 parseAnyChar :: String -> Parser Char
   32 parseAnyChar [] = Parser $ const Nothing
   33 parseAnyChar (a:ax) = Parser $ \input ->
   34   case input of
   35     (b:bx) | a == b -> Just (a, bx)
   36     (_:_) -> runParser (parseAnyChar ax) input
   37     [] -> Nothing
   38 
   39 parseOr :: Parser a -> Parser a -> Parser a
   40 parseOr p1 p2 = Parser $ \s ->
   41   case runParser p1 s of
   42     Just (a, s') -> Just (a, s')
   43     Nothing -> runParser p2 s
   44 
   45 parseAnd :: Parser a -> Parser b -> Parser (a, b)
   46 parseAnd p1 p2 = Parser $ \s ->
   47   case runParser p1 s of
   48     Just (a, s') ->
   49       case runParser p2 s' of
   50         Just (b, s'') -> Just ((a, b), s'')
   51         Nothing -> Nothing
   52     Nothing -> Nothing
   53 
   54 parseAndWith :: (a -> b -> c) -> Parser a -> Parser b -> Parser c
   55 parseAndWith f p1 p2 = Parser $ \s ->
   56   case runParser p1 s of
   57     Just (a, s') ->
   58       case runParser p2 s' of
   59         Just (b, s'') -> Just (f a b, s'')
   60         Nothing -> Nothing
   61     Nothing -> Nothing
   62 
   63 parseMany :: Parser a -> Parser [a]
   64 parseMany p = Parser $ \s ->
   65   case runParser p s of
   66     Just (a, s') ->
   67       case runParser (parseMany p) s' of
   68         Just (as, s'') -> Just (a : as, s'')
   69         Nothing -> Just ([a], s')
   70     Nothing -> Just ([], s)
   71 
   72 parseSome :: Parser a -> Parser [a]
   73 parseSome p = Parser $ \s ->
   74   case runParser p s of
   75     Just (a, s') ->
   76       case runParser (parseMany p) s' of
   77         Just (as, s'') -> Just (a : as, s'')
   78         Nothing -> Nothing
   79     Nothing -> Nothing
   80 
   81 parseUInt :: Parser Int
   82 parseUInt = Parser $ \s ->
   83   case runParser (parseSome (parseAnyChar ['0' .. '9'])) s of
   84     Just (as, s') -> Just (read as, s')
   85     Nothing -> Nothing
   86 
   87 parseInt :: Parser Int
   88 parseInt = Parser $ \s ->
   89   case runParser (parseOr (parseChar '-') (parseChar '+')) s of
   90     Just (c, s') ->
   91       case runParser parseUInt s' of
   92         Just (i, s'') -> Just (if c == '-' then -i else i, s'')
   93         Nothing -> Nothing
   94     Nothing -> runParser parseUInt s
   95 
   96 parsePair :: Parser a -> Parser (a, a)
   97 parsePair p = Parser $ \s ->
   98   case runParser (parseAndWith (,) (parseChar '(') (parseAndWith (,) p (parseAndWith (,) (parseChar ' ') p))) s of
   99     Just ((_, (a, (_, b))), ')' : s') -> Just ((a, b), s')
  100     _ -> Nothing
  101 
  102 parseList :: Parser a -> Parser [a]
  103 parseList p = Parser $ \s ->
  104   case runParser (parseChar '(') s of
  105     Just (_, s') -> parseList' p [] s'
  106     Nothing -> Nothing
  107   where
  108     parseList' :: Parser a -> [a] -> String -> Maybe ([a], String)
  109     parseList' parser acc s' =
  110       case runParser parser s' of
  111         Just (x, s'') ->
  112           case runParser (parseChar ' ') s'' of
  113             Just (_, s''') -> parseList' parser (acc ++ [x]) (dropWhile (== ' ') s''')
  114             Nothing ->
  115               case runParser (parseChar ')') s'' of
  116                 Just (_, s''') -> Just (acc ++ [x], s''')
  117                 Nothing -> Nothing
  118         Nothing ->
  119           case runParser (parseChar ')') s' of
  120             Just (_, s'') -> Just (acc, s'')
  121             Nothing -> Nothing

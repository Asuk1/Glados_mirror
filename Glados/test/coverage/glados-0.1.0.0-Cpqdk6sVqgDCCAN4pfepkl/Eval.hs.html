   16 data Result = Value Int | Bool String | Err String
   17 
   18 instance Eq Result where
   19     (Value x) == (Value y) = x == y
   20     (Bool s1) == (Bool s2) = s1 == s2
   21     (Err e1) == (Err e2) = e1 == e2
   22     _ == _ = False
   23 
   24 instance Show Result where
   25     show (Value x) = "Value " ++ show x
   26     show (Bool s) = "Bool " ++ show s
   27     show (Err s) = "Err " ++ show s
   28 
   29 
   30 getValue :: Ast -> Env -> Result
   31 getValue (AstInteger a) _ = Value a
   32 getValue (AstSymbol s) env =
   33     case lookup s env of
   34         Just value -> getValue value env
   35         Nothing -> Err ("Symbol '" ++ s ++ "' not found in the environment.")
   36 getValue _ _ = Err "Error: Unsupported expression type"
   37 
   38 add :: [Ast] -> Env -> Result
   39 add [a, b] env =
   40     case (getValue a env, getValue b env) of
   41         (Value x, Value y) -> Value (x + y)
   42         (Err errA, _) -> Err ("Error in 'a': " ++ errA)
   43         (_, Err errB) -> Err ("Error in 'b': " ++ errB)
   44         _ -> Err "Error: Addition requires two integer values."
   45 add _ _ = Err "Error in add: Insufficient arguments"
